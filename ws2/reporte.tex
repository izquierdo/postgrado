\newcommand{\tablaTripletas}[2]
{
\begin{tabular}{l|c}
 & Número de tripletas \\
 \hline
 Antes & #1 \\
 Después & #2 \\
\end{tabular}
}

\newcommand{\tablaPredicados}[1]
{
\begin{tabular}{l|c|c}
Predicado & Antes & Después \\
\hline
#1
\end{tabular}
}

\newcommand{\tablaInstancias}[1]
{
\begin{tabular}{l|c|c|c}
Clase & Antes & Después \\
\hline
#1
\end{tabular}
}
